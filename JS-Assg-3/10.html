Fibonacci Series

WAF: To generate fibonacci series upto user entered number eg: if user enters 6. output: 1 1 2 3 5 8 
				
function fibonacci(digit) {
	var firstNumb = 1, secNumb = 1; //declaring 1st & 2nd number
	var fibo = [firstNumb, secNumb]; //making array for series & putting 1st & 2nd number

	//hnding if user enters 1
	if( digit == 1 ){
		fibo = [firstNumb];
	}
	//if user enters more than 2
	else if( digit > 2 ){
		var newNumb;
		for (var i = 3; i <= digit; i++) { //running for loop from 3rd position to 'digit'
			newNumb = firstNumb + secNumb; // new number is addition of 1st & 2nd number
			fibo.push(newNumb); //pushing new value in fibo series array

			//saving the 2nd post in 1st post, then new numb in 2nd pos and loop ontinues
			firstNumb = secNumb; 
			secNumb = 	newNumb;	
		}
	}
	return fibo;
}

do{
	var digit = prompt("Enter how many numbers you want in the Fibonacci series(greater than 0):", '5');
}while( isNaN(digit) || digit <= 0 );


alert( 'Fibonacci Series: ' + fibonacci(digit) );
