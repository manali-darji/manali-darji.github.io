Re-write code using Ternary operator

Rewrite below function using ternary operator
				
function checkAge(age) {
  if (age > 18) {
    return true;
  } else {
    return confirm('Did parents allow you?');
  }
}
ANSWER:
				
function checkAge(age) {
	if( isNaN(age) ){
		return "Age should be number";
	}else if( age < 0 || age > 100){
		return "Age should be beween 1 to 100";
	}
	else{
		return (age > 18) ? true : confirm('Did parents allow you?');
	} 
}

var age = prompt('Enter age to check', '10');
alert( checkAge(age) );
