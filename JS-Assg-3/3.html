Minimum from 2 using Ternary Operator

WAF to find min value from 2 numbers using ternary operators 
				
function min(a,b) {
	if( isNaN(a) || isNaN(b) ){
		return "Values should be numbers";
	}else if( a == b){
		return "Numbers are equal";
	}else{
		return "Minimum from "+ a + " & " + b + " is " + ( ( a < b ) ? a : b ); 
	}
}

alert( min(8, 15) );
alert( min('eight', 15) );
alert( min(8, 8) );
