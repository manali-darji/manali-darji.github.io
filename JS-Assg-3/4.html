Minimum from 2 using Ternary Operator

WAF to find prime numbers from 1-100
				
function isPrime(numb) {
	var isPrime = false;
	for (var i = 2; i <= numb; i++) {
		if( (numb % i == 0) && ( numb != i) ){
			isPrime = false;
			break;
		}else{
			isPrime = true;
		}
	}

	if(isPrime){return numb;}
	else{return false;}
}

var primeNumbs = [];
for (var i = 1; i <= 100; i++) {
	if( isPrime(i) ){ primeNumbs.push(i); }
}
alert( 'Prime Numbers: ' + primeNumbs );
