Multiply two or multile Numbers

WAF to multiply 2/multiple numbers. (use function parameters/arguments)
				
function multiply_numbs(){
	var ans = 1;
	for (var i = 0; i < arguments.length; i++) {
		ans *= arguments[i];
	}
	return ans;
}

alert( multiply_numbs(2,5) );
alert( multiply_numbs(10,10,10) );
alert( multiply_numbs(10,10,10,10,10) );
