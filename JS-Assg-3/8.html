Alert today's Day. Also print a message saying “It is a weekend” if Saturday or Sunday!

Execute Code
							
function AlertTodaysDay(){
	var todayDate = new Date();
	var day = todayDate.getDay();

	switch(day){
		case 0:
		case 6:
		alert('Its a weekend, yaaayy !');
		break;

		case 1:
		alert('Oh God Monday is back');
		break;

		case 2:
		alert("What!!!! It's only Tuesday ?!!");
		break;

		case 3:
		alert('Wednesday, OK Half week done.');
		break;

		case 4:
		alert('Thursday, almost there!');
		break;

		case 5:
		alert('YAY, FRIDAY is here !!');
		break;
	}	
}
AlertTodaysDay();

function changeBgColorAccToTime(){
	var date = new Date();
	var hours = date.getHours();

	if( hours >= 0 && hours < 15){
		bodyColor = '#d8d853';
	}else if( hours >= 15 && hours < 19){
		bodyColor = '#375385';
	}else{
		bodyColor = '#000';
	}

	document.getElementsByTagName('body')[0].style.backgroundColor = bodyColor;
}
changeBgColorAccToTime();
