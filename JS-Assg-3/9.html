function isPalindrome(word) {
	// if null value is given
	if( !word ){ return false; }
	// setting counter  
	var count = 0; 

	// find the middle position of the word to use in for loops
	var midOfWord = Math.floor(word.length/2); 

	// setting a variable 'check' which should be equal to 'count' in order to be a Palindrome string
	// Handling the value of 'check' for even/odd number of letters in the word
	var check = (word.length % 2) ? (midOfWord+1) : midOfWord; 

	// used 2 variables in a single for loop
	// 'i' starts from 0th position to midOfWord 
	// 'j' simultaneously starts from last position to midOfWord 
	for (var i = 0, j = word.length-1; i <= midOfWord, j >= midOfWord; i++, j--) {
		// if the letters are same, increasing the count
		if( word[i] == word[j] ){
			count++;
		}			
	}
	//checking 'count' with 'check' variable and returning value accordingly
	return (count == check) ? true : false;
}

var word = prompt("Enter string to check if Palindrome or not:", "level");

if( isPalindrome(word) ){alert(word + " is Palindrome");}
else{alert(word + " is not a Palindrome");}
